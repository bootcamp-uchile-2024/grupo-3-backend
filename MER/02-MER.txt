Table usuarios{
  id integer [primary key]
  contrasena varchar
  id_tipo_usuario integer
  nombre varchar
  apellido varchar
  nombre_usuario varchar
  email varchar
  telefono varchar
  genero varchar
  rut varchar
  fecha_nacimiento date
}


Table direcciones{
  id integer
  id_usuario integer
  comuna varchar
  calle varchar
  numero varchar
  departamento varchar
  referencia varchar
}


Table tipo_usuarios{
  id integer
  tipo varchar
}


Table usuarios_medios_pagos{
  id_usuario integer [primary key]
  id_medio_pago integer [primary key]
  es_preferido bool
}


Table productos{
  id integer [primary key]
  SKU varchar
  nombre varchar
  id_categoria integer
  precio integer
  descripcion varchar
  imagen varchar
  cantidad integer
  unidades_vendidas integer
  puntuacion float
  ancho integer
  alto integer
  largo integer
  peso integer
}


Table etiquetas{
  id integer [primary key]
  etiqueta varchar
}


Table productos_etiquetas{
  id_producto integer [primary key]
  id_etiqueta integer [primary key]
}


Table pedidos{
  id integer [primary key]
  id_usuario integer
  fecha_creacion date
  id_medio_pago integer
  id_estado integer
  id_tipo_despacho integer
  id_carro integer
  fecha_entrega date
}


Table carros{
  id_usuario integer
  id integer [primary key]
  fecha_creacion date
  fecha_cierre date  
}


Table pagos{
  id integer [primary key]
  id_medio_pago integer
  id_pedido integer
  fecha datetime
  monto integer
}


Table medio_pago{
  id integer [primary key]
  nombre varchar
  habilitado bool
}

Table tipo_despacho{
  id integer [primary key]
  tipo varchar
}


Table estados_pedido{
  id integer [primary key]
  estado varchar
}


Table categorias{
  id integer
  categoria varchar
}


Table carros_productos{
  id_carro integer [primary key]
  id_producto integer [primary key]
  cantidad_producto integer
}

Table marcas{
  id integer [primary key]
  nombre varchar
}
//Atributos de categorias
Table plantas{
  id_producto integer [primary key]  
  id_especie integer
  pet_friendly bool
  id_tipo_riego integer
  id_fotoperiodo integer
  tolerancia_temperatura integer
  ciclo bool
  altura varchar
  id_habito_crecimiento integer
  id_color integer
}
Table plantas_especies{
  id integer [primary key]
  especie varchar
}
Table plantas_fotoperiodo{
  id integer [primary key]
  tipo_fotoperiodo varchar
}
Table plantas_tipo_riego{
  id integer [primary key]
  tipo_riego varchar
}
Table colores_productos{
  id integer [primary key]
  color varchar
}
Table plantas_habito_crecimiento{
  id integer [primary key]
  crecimiento varchar
}


Table maceteros{
  id_producto integer [primary key]
  id_marca integer
  id_tipo_macetero integer
  material varchar
  forma varchar
  diametro integer
}
Table tipo_maceteros{
  id integer [primary key]
  tipo varchar
}

Table insumos{
  id_producto integer [primary key]
  id_tipo_insumo integer
  id_marca integer
}
Table tipo_insumos{
  id integer [primary key]
  tipo_insumo varchar
}

Table accesorios{
  id_producto integer [primary key]
  id_marca integer
  id_tipo_accesorio integer
  id_color integer
}
Table tipo_accesorios{
  id integer [primary key]
  tipo varchar
}
